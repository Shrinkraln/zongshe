1. 在同一个容器中使用两个终端
```bash
#启动agent
docker run -it -d --name mragent \
    --net=host \
    --privileged \
    -v /dev/shm:/dev/shm \
    microros/micro-ros-agent:humble udp4 --port 8888 -v4
#启动rosb
docker run -it -d --name rosb \
    --net=host \
    --restart unless-stopped \
    rosb \
    ros2 launch rosbridge_server rosbridge_websocket_launch.xml topic_polling_interval:=100
#可以执行指令
docker exec -it rosb bash
#这个需要激活ros环境
source /opt/ros/humble/setup.bash
```
